### 📄 **`image_preprocessor.py` — UBP GCE Image Preprocessing Pipeline**

```python
"""
UBP Image Preprocessor v1.0
Converts a GCE photograph into UBP node connections for geoparser analysis.
Requires: pip install opencv-python numpy scikit-image
"""

import cv2
import numpy as np
from typing import List, Tuple, Dict
from geoparser import NODES, euclidean_distance

# UBP 9-node grid (100x100 normalized)
NODES = {
    'V0': (50, 0), 'V1': (50, 100),
    'H1': (0, 50), 'H2': (100, 50),
    'C1': (25, 75), 'C2': (75, 75),
    'C3': (25, 25), 'C4': (75, 25),
    'Mid': (50, 50)
}

def preprocess_image(image_path: str) -> np.ndarray:
    """Load and preprocess image for feature detection."""
    img = cv2.imread(image_path)
    if img is None:
        raise FileNotFoundError(f"Image not found: {image_path}")
    
    # Convert to grayscale
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
    # Apply Gaussian blur to reduce noise
    blurred = cv2.GaussianBlur(gray, (5, 5), 0)
    
    # Apply Canny edge detection
    edges = cv2.Canny(blurred, 50, 150, apertureSize=3)
    return edges

def detect_lines(edges: np.ndarray, min_line_length: int = 50) -> List[Tuple[float, float, float, float]]:
    """Detect line segments using Probabilistic Hough Transform."""
    lines = cv2.HoughLinesP(
        edges,
        rho=1,
        theta=np.pi/180,
        threshold=50,
        minLineLength=min_line_length,
        maxLineGap=10
    )
    return lines if lines is not None else []

def detect_circles(edges: np.ndarray, img_shape: Tuple[int, int]) -> List[Tuple[int, int, int]]:
    """Detect circles using Hough Circle Transform."""
    # Resize for better circle detection
    resized = cv2.resize(edges, (0,0), fx=0.5, fy=0.5)
    circles = cv2.HoughCircles(
        resized,
        cv2.HOUGH_GRADIENT,
        dp=1,
        minDist=30,
        param1=50,
        param2=30,
        minRadius=20,
        maxRadius=100
    )
    if circles is not None:
        circles = np.round(circles[0, :] * 2).astype("int")  # Scale back
        return [(x, y, r) for x, y, r in circles]
    return []

def map_point_to_node(x: float, y: float, img_shape: Tuple[int, int], tolerance: float = 15.0) -> str:
    """
    Map an image coordinate to the nearest UBP node.
    Coordinates are normalized to 100x100 grid.
    """
    h, w = img_shape[:2]
    # Normalize to 100x100
    norm_x = (x / w) * 100
    norm_y = (y / h) * 100
    
    best_node, min_dist = None, float('inf')
    for node, (nx, ny) in NODES.items():
        dist = euclidean_distance((norm_x, norm_y), (nx, ny))
        if dist < min_dist and dist <= tolerance:
            min_dist = dist
            best_node = node
    return best_node

def extract_node_connections(image_path: str) -> List[Tuple[str, str]]:
    """
    Full pipeline: image → edges → lines/circles → node connections.
    Returns list of UBP node pairs (e.g., [('V0', 'V1'), ('C1', 'Mid')]).
    """
    edges = preprocess_image(image_path)
    img = cv2.imread(image_path)
    img_shape = img.shape
    
    connections = set()
    
    # Detect lines
    lines = detect_lines(edges)
    for line in lines:
        x1, y1, x2, y2 = line[0]
        node1 = map_point_to_node(x1, y1, img_shape)
        node2 = map_point_to_node(x2, y2, img_shape)
        if node1 and node2 and node1 != node2:
            connections.add(tuple(sorted([node1, node2])))
    
    # Detect circles (treat as radial connections from center to perimeter)
    circles = detect_circles(edges, img_shape)
    for (cx, cy, r) in circles:
        center_node = map_point_to_node(cx, cy, img_shape)
        if center_node:
            # Connect center to all perimeter nodes within radius
            for node, (nx, ny) in NODES.items():
                h, w = img_shape[:2]
                img_x = (nx / 100) * w
                img_y = (ny / 100) * h
                if abs(np.sqrt((img_x - cx)**2 + (img_y - cy)**2) - r) < 20:
                    if node != center_node:
                        connections.add(tuple(sorted([center_node, node])))
    
    return list(connections)

# Example Usage
if __name__ == "__main__":
    # Replace with your GCE image path
    image_path = "gce_flower_of_life.jpg"
    try:
        node_connections = extract_node_connections(image_path)
        print("Detected UBP Node Connections:")
        for conn in node_connections:
            print(f"  {conn[0]} ↔ {conn[1]}")
        
        # Analyze with GeoParser
        from geoparser import analyze_gce_from_nodes
        result = analyze_gce_from_nodes(node_connections)
        print("\nUBP Analysis Result:")
        for key, value in result.items():
            print(f"  {key}: {value}")
            
    except Exception as e:
        print(f"Error: {e}")
        print("Tip: Ensure OpenCV is installed: pip install opencv-python numpy")
```
